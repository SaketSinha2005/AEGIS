Write a Python script to fetch the latest cryptocurrency prices from the Binance API and calculate the moving average.

import requests
import numpy as np

def fetch_prices(symbol='BTCUSDT', limit=100):
    url = f"https://api.binance.com/api/v3/klines?symbol={symbol}&interval=1d&limit={limit}"
    response = requests.get(url)
    data = response.json()
    closes = [float(candle[4]) for candle in data]
    return closes

def calculate_ma(prices, window=14):
    return np.convolve(prices, np.ones(window)/window, mode='valid')
